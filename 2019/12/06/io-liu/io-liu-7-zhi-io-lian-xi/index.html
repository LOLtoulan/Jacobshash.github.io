/**
 * flush方法作用，刷新缓存，可以继续写入
 * close();有同样效果，但是不能继续写入数据
 */
public class test1 {
    public static void main(String[] args) throws IOException {

        //多态形式
        OutputStream outputStream = new FileOutputStream("LOLtoulan\\b.txt");
        //非多态
        //FileOutputStream fileOutputStream = new FileOutputStream("LOLtoulan\\a.txt");
        //FileInputStream fileInputStream = new FileInputStream("LOLtoulan\\b.txt");

        //第一种
        byte[] bytes = {'a','b','c','\n'};
        outputStream.write(bytes);
        outputStream.flush();

        //第二种
        for (int i = 0; i < 10; i++) {
            outputStream.write(("hello" + i + "\n").getBytes());
        }
        outputStream.flush();

        //第三种
        BufferedOutputStream bufferedOutputStream = new BufferedOutputStream(outputStream);
        byte[] bytes1 = {'a','b','c','\n'};
        bufferedOutputStream.write(bytes1);
        bufferedOutputStream.flush();

        //第四种
        for (int i = 0; i < 10; i++) {
            bufferedOutputStream.write(("world" + i + "\n").getBytes());
        }
        bufferedOutputStream.flush();

        //第五种
        bufferedOutputStream.write('a');
        bufferedOutputStream.write('\n');
        bufferedOutputStream.flush();
        //第六种 ascii值的a
        bufferedOutputStream.write(97);
        bufferedOutputStream.flush();

        //第七种
        outputStream.write('b');
        outputStream.write('\n');

        //第八种
        outputStream.write(98);
        outputStream.flush();

        Writer writer = new OutputStreamWriter(outputStream);
        //第九种
        writer.write("我是一个字符串，由字符流写入");
        //第十种
        writer.append('a');
        writer.append("ddssdfsdfsd");
        writer.flush();

        //第十一种
        OutputStreamWriter outputStreamWriter = new OutputStreamWriter(outputStream);
        outputStreamWriter.write("我是由字符流OutputStreamWriter输入");

        //第十二种
        String[] s = {"ass","bss","我是一个字符串数组","由OutputStreamWriter写入"};
        for (String ss : s) {
            outputStreamWriter.write(ss);
            //第十三种
            outputStreamWriter.write(ss,0,2);
        }
        outputStreamWriter.flush();

        //第十四种
        BufferedWriter bufferedWriter = new BufferedWriter(writer);
        bufferedWriter.write("我是由字符流BufferedWriter输入");
        bufferedWriter.write("我是由字符流BufferedWriter输入",0,15);
        //换行输入
        bufferedWriter.newLine();

        //第十五种
        String[] s1 = {"ass","bss","我是一个字符串数组","由OutputStreamWriter写入"};
        for (String ss : s1) {
            bufferedWriter.write(ss);
            bufferedWriter.newLine();
        }
        bufferedWriter.flush();
        //第十六种
        writer.write(99);
        writer.flush();
    }
}

public class Test6 {
    public static void main(String[] args) throws IOException {
        ArrayList<Student> arrayList = new ArrayList<>();
        BufferedReader bufferedReader = new BufferedReader(new FileReader("LOLtoulan\\e.txt"));

        String len = null;
        String sb;
        while ((len = bufferedReader.readLine()) !=null) {
            //学生对象必须定义在while循环里面否则会出现bug，因为如果定义在外面，无论你怎样操作都始终操作的是一个对象，且当while循环做完时，这时最后一个对象一直在缓存中，所以每次取值都获取的是最后那个对象。
            Student student = new Student();
            sb = len;
            String[] s = sb.split(":");
            student.setName(s[0]);
            student.setAge(Integer.parseInt(s[1]));
            student.setSex(s[2]);
            arrayList.add(student);
        }

        for (Student s : arrayList) {
            System.out.println(s.getName()+":"+s.getAge()+":"+s.getSex());
        }
        bufferedReader.close();
    }
}

public class StudentDTO {
    private String name;
    private int Chinese;
    private int Math;
    private int English;

    public StudentDTO(String name, int chinese, int math, int english) {
        this.name = name;
        Chinese = chinese;
        Math = math;
        English = english;
    }

    public StudentDTO() {
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getChinese() {
        return Chinese;
    }

    public void setChinese(int chinese) {
        Chinese = chinese;
    }

    public int getMath() {
        return Math;
    }

    public void setMath(int math) {
        Math = math;
    }

    public int getEnglish() {
        return English;
    }

    public void setEnglish(int english) {
        English = english;
    }

    @Override
    public String toString() {
        return "Student{" +
                "name='" + name + '\'' +
                ", Chinese=" + Chinese +
                ", Math=" + Math +
                ", English=" + English +
                '}';
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof StudentDTO)) return false;
        StudentDTO student = (StudentDTO) o;
        return getChinese() == student.getChinese() &&
                getMath() == student.getMath() &&
                getEnglish() == student.getEnglish() &&
                getName().equals(student.getName());
    }

    @Override
    public int hashCode() {
        return Objects.hash(getName(), getChinese(), getMath(), getEnglish());
    }

    //计算总分，写在这个地方可提高方法复用性
    public int getSum(){
        int sum = this.Chinese + this.English + this.Math;
        return sum;
    }
}

十二、通过Reader自定义BufferedRreader的readerLine()方法

public class MyBufferedReader {
    private Reader reader;

    public MyBufferedReader(Reader reader) {
        this.reader = reader;
    }

    /**
     * 自定义 BufferedReader 的 readerLine() 功能
     * @return
     * @throws IOException
     */
    public String readerLine() throws IOException {
        /*
        我们通过查API发现Reader 的read()方法有两种
        一种是读取一个字节的：public int read()
        一种是读取一个字节数组的：public int read(char[] cbuf)
        但是我们发现无法定义数组长度，因为无论定多大，都可能会不够，也有可能会太大而浪费内存
        但是我们如果定义为一次读取一个字节的方法，当我们读取下一个字节时上一个字节就会消失，
        所以我们必须定义字符缓冲区，有三个可选项，数组(不知道定义多大)，集合(每次都要调用add方法，
        之后还要转换为字符串)，字符串缓冲区，所以我们简单的考虑美酒定义为字符缓冲区
        且为StringBuffer

         */
        StringBuffer stringBuffer = new StringBuffer();
        int ch = 0;
        while ((ch = reader.read()) != -1) {
            if (ch == '\r') {
                continue;
            }
            if (ch == '\n') {
                return stringBuffer.toString();
            } else {
                stringBuffer = stringBuffer.append((char)ch);
            }
        }
        if (stringBuffer.length() > 0) {
            return stringBuffer.toString();
        }

        return null;
    }

    public void close() throws IOException {
        this.reader.close();
    }
}
public class MyBufferedReaderTest {
    public static void main(String[] args) throws IOException {

        MyBufferedReader myBufferedReader = new MyBufferedReader(new FileReader("LOLtoulan\\newK.txt"));

        String  len = null;
        while ((len = myBufferedReader.readerLine()) != null) {
            System.out.println(len);
        }
    }
}
